def build_model_1_1(numhiddenlayers=3):
    model = models.Sequential(name='model_1_1')
    #input layer
    model.add(layers.Reshape((3072,), input_shape=(32,32,3)))
    
    number_of_nodes = 3072
    #add hidden layers (half as many nodes) with rectified linear units
    for i in range(numhiddenlayers):
      number_of_nodes /= 2
      model.add(layers.Dense(number_of_nodes, activation='relu'))
    
    #output layer should output the predicted label by percentage
    model.add(layers.Dense(10, activation='softmax'))

    #opt = optimizers.SGD(learning_rate=0.01, momentum=0.5, nesterov=True)
    opt = optimizers.Adagrad(learning_rate=0.01)
    model.compile(optimizer=opt,
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])
    return model

def build_model_1_3(numhiddenlayers=3):
    model = models.Sequential(name='model_1_3')
    #input layer
    model.add(layers.Reshape((1024,), input_shape=(32,32,1)))
    
    dropout_rate = 0.1
    number_of_nodes = 3072
    #add hidden layers (half as many nodes) with rectified linear units
    for i in range(numhiddenlayers):
      number_of_nodes /= 2
      layer_dropout = dropout_rate / (i+1)
      #because we use dropout we need more nodes in each layer
      nodes = number_of_nodes * (1 + layer_dropout)
      model.add(layers.Dense(nodes, activation='relu', bias_regularizer=tf.keras.regularizers.l2(l=0.01)))
      model.add(layers.Dropout(rate=layer_dropout))
    
    #output layer should output the predicted label by percentage
    model.add(layers.Dense(10, activation='softmax'))

    opt = optimizers.Adagrad(learning_rate=0.01)
    model.compile(optimizer=opt,
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])
    return model

Answer 2.1, ConvNet (7p)¶

  • Build a sequential ConvNet, get at least 90% accuracy.
    • Conv2D layers, input shape (32,32,1)
    • Consider using blocks of Conv2D layers with maxpooling in between
    • Use zero-padding in Conv2D layers because the images are small
    • At least one dense hidden layer and dense output layer (sigmoid)
    • Consider adding batch normalization, dropout
    • Consider small dropout rates in beginning, larger dropout rates later in network
    • Consider other optimizers and tuning learning rate, momentum,...
  • Explain what you did and why (800 chars). Is it working well?
    • Should be quite elaborate and sensible
I've created a CNN with 3 conv layers with increasing amount of filters 32 to 64 to 128, the intuition behind this was that every subsequent conv layer can combine multiple features from the previous one which results in more posibilities. I've added Maxpooling (to create an abstracted form of the input), batchnormalization (to improve the learning speed) and dropout (to reduce overfitting) to these conv layers. After the conv layers we flatten the input and feed it trough a dense layer and finally using softmax with categorical crossentropy like before. I've added l2 regularization to the conv and dense layer to prevent overfitting (much lower training loss than validation loss). The model performs fairly well because the validation it's not under or overfitting within 15 epochs. (length: 792)

def build_model_2_1(numhiddenlayers=3, modelname='model_2_1'):
    model = models.Sequential(name=modelname)
    #input layer convolutional layer
    model.add(layers.Conv2D(32, (5, 5), activation='relu', input_shape=(32, 32, 1), kernel_regularizer=tf.keras.regularizers.l2(l=1e-4)))
    model.add(layers.MaxPooling2D((2, 2)))
    model.add(layers.BatchNormalization())
    model.add(layers.Dropout(rate=0.1))
    # conv + maxpool + normalize
    model.add(layers.Conv2D(64, (4, 4), activation='relu', kernel_regularizer=tf.keras.regularizers.l2(l=1e-4)))
    model.add(layers.MaxPooling2D((2, 2)))
    model.add(layers.BatchNormalization())
    model.add(layers.Dropout(rate=0.1))
    # conv + maxpool + normalize
    model.add(layers.Conv2D(128, (3, 3), activation='relu', kernel_regularizer=tf.keras.regularizers.l2(l=1e-4)))
    model.add(layers.MaxPooling2D((2, 2)))
    model.add(layers.BatchNormalization())
    model.add(layers.Dropout(rate=0.1))
    
    #convert back to single dimension
    model.add(layers.Flatten())
    #dense layer with a dropout
    model.add(layers.Dense(64, activation='relu', kernel_regularizer=tf.keras.regularizers.l2(l=1e-4)))
    #output layer should output the predicted label by percentage
    model.add(layers.Dense(10, activation='softmax'))

    opt = optimizers.Adagrad(learning_rate=0.02)
    model.compile(optimizer=opt,
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])
    return model

def build_model_4_1(modelname='model_4_1'):
  vgg_conv = VGG16(weights='imagenet', include_top=False, input_shape=(32, 32, 3))
  #freeze every layer except specified
  for layer in vgg_conv.layers:
    layer.trainable = False
  
  # Create new model
  model = models.Sequential(name=modelname)
        # Add the vgg convolutional base model
  model.add(vgg_conv)
  # Add new layers
  model.add(layers.Flatten())
  model.add(layers.Dense(256, activation='relu'))
  model.add(layers.Dense(10, activation='softmax'))
  
  opt = optimizers.Adagrad(learning_rate=0.02)
  model.compile(optimizer=opt,
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])
  return model
